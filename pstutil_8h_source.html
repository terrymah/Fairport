00117 inline std::vector<fairport::byte> fairport::calculate_entry_id(const shared_db_ptr& db, node_id nid)
00118 {
00119     using namespace std;
00120 
00121     // This function doesn't know how to handle anything other than vanilla
00122     // PSTs.  OSTs, for example, require a more complex calculation.
00123     if (!db->is_pst())
00124         throw key_not_found<prop_id>(0x0fff);
00125 
00126     // A MAPI entry id contains 4 leading 0 bytes, the data store ID, and
00127     // the node ID (in little-endian byte order).
00128     vector<byte> entry_id(4, 0);
00129 
00130     node store(db->lookup_node(nid_message_store));
00131     property_bag store_props(store);
00132     vector<byte> store_id(store_props.read_prop<vector<byte> >(0x0ff9));
00133     copy(store_id.begin(), store_id.end(),
00134          insert_iterator<vector<byte> >(entry_id, entry_id.end()));
00135 
00136     for (size_t i = 0; i < sizeof(node_id); ++i)
00137         entry_id.push_back((nid >> 8*i) & 0xff);
00138 
00139     return entry_id;
